CREATE TABLE REUNIAO (ID  SERIAL NOT NULL, ASSUNTO VARCHAR(255), DATAHORA VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE PARTICIPANTE (ID  SERIAL NOT NULL, DTYPE VARCHAR(31), EMAIL VARCHAR(255), NOME VARCHAR(255), EMPRESA VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE REUNIAO_PARTICIPANTE (reunioes_ID INTEGER NOT NULL, participantes_ID INTEGER NOT NULL, PRIMARY KEY (reunioes_ID, participantes_ID))
ALTER TABLE REUNIAO_PARTICIPANTE ADD CONSTRAINT FK_REUNIAO_PARTICIPANTE_reunioes_ID FOREIGN KEY (reunioes_ID) REFERENCES REUNIAO (ID)
ALTER TABLE REUNIAO_PARTICIPANTE ADD CONSTRAINT FK_REUNIAO_PARTICIPANTE_participantes_ID FOREIGN KEY (participantes_ID) REFERENCES PARTICIPANTE (ID)
